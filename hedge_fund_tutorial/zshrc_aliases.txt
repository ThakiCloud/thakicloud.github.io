# 헤지펀드 시계열 모델링 환경 알리아스
# ~/.zshrc에 추가하세요

# 환경 설정
alias hedge_setup='cd /Users/hanhyojung/work/thakicloud/thakicloud.github.io/hedge_fund_tutorial'
alias hedge_env='source hedge_fund_env/bin/activate'
alias hedge_activate='hedge_setup && hedge_env'

# 테스트 실행
alias test_garch='python test_garch_simple.py'
alias run_hedge_test='hedge_activate && test_garch'

# 개발 도구
alias hedge_jupyter='hedge_activate && jupyter notebook'
alias hedge_install='pip install'
alias hedge_list='pip list | grep -E "(numpy|pandas|scipy|sklearn|torch|ray)"'

# 시스템 정보
alias sys_info='echo "CPU: $(sysctl -n machdep.cpu.brand_string)" && echo "Cores: $(sysctl -n hw.ncpu)" && echo "Memory: $(echo "$(sysctl -n hw.memsize) / 1024 / 1024 / 1024" | bc) GB"'

echo "🔧 zshrc 알리아스 설정 완료!"
echo "사용법:"
echo "  hedge_activate  # 환경 활성화"
echo "  run_hedge_test  # GARCH 테스트 실행"
echo "  sys_info        # 시스템 정보 확인"
